import random
import numpy as np

actions = ['u','r','d','l']
qline = {'u':1.2, 'r':-2.1, 'd':-24.5, 'l':27}
epsilon = 0.3 # 0.3

def choose_action(a, q, eps=None):
    action = max(q, key=lambda x: q[x])
    if eps is not None:
        policy = np.ones(len(q)) * eps / len(q)
        policy[a.index(action)] = 1 - eps + (eps / len(q))
        action = np.random.choice(a, p=policy) 
    return action
